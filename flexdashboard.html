data("instacart")
instacart = 
  instacart %>% 
  janitor::clean_names() %>% 
  drop_na()

Chart A-Number of items odered in aisle

instacart %>%
  count(aisle) %>%
  rename(count = n) %>%
  filter(count > 10000) %>%
  mutate(
    aisle = factor(aisle),
    aisle = forcats::fct_reorder(aisle, count)
  ) %>%
  plot_ly(x = ~count, y = ~aisle, type = "bar") %>% 
  layout(yaxis = list(title = "Number of items odered in aisle"))
